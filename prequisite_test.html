# 1) List Operations and Basic Statistics

l=[i for i in range(1,21)]
print(l)

even_sum=0
for j in l:
    if j%2==0:
        even_sum+=j
print("Sum of all even numbers in the list:",even_sum)

product=1
for j in l:
    if j%2!=0:
        product*=j
print("product of all odd numbers in the list:",product)

mean = sum(l)/len(l)
print("Mean:",mean)

sorted_list = sorted(l)
n = len(sorted_list)
if n % 2 == 1:
    median = sorted_list[n // 2]
else:
    median = (sorted_list[n // 2 - 1] + sorted_list[n // 2]) / 2
print("Median:",median)

mean_diff_squared = [(x - mean) ** 2 for x in l]
sd = (sum(mean_diff_squared) / (n-1)) ** 0.5
print("Standard deviation:",round(sd,2))

import matplotlib.pyplot as plt
import numpy as np


n=int(input("Enter the no of students:" ))
d={}
for _ in range (n):
    student=str(input())
    marks=float(input())
    if student not in d.keys():
        d.update({student:marks})
    else:
        d[student]+=marks
        d.update({student:marks})
        
lst1=[i for i in d.values()]
average=sum(lst1)/len(lst1)
print("Average marks:",round(average,2))

print("Students scoring above average are:")
for a,b in d.items():
    if b>average:
        print(a)

y=[i for i in d.values()]

plt.bar(x,y, label='Marks distribution',width=0.4)

#VENN DIAGRAMS

import matplotlib.pyplot as plt
from matplotlib_venn import venn2

l1=eval(input("List1:"))
l2=eval(input("List2:"))

s1=set(l1)
s2=set(l2)

intersec=s1.intersection(s2)
un=s1.union(s2)
sym_dif=s1.symmetric_difference(s2)

print("Intersection:",list(intersec))
print("Union:",list(un))
print("Symmetric difference:",list(sym_dif))

plt.figure(figsize=(6, 6))
venn=venn2([s1, s2],set_labels=('Set 1', 'Set 2'))
plt.title("Venn Diagram of two sets")
plt.show()

#21) Basic statstics 

import matplotlib.pyplot as plt

data = eval(input("Enter list of numbers:"))

mean = sum(data)/len(data)
print("Mean:",round(mean,2))

sorted_data = sorted(data)
n = len(sorted_data)
if n % 2 == 1:
    median = sorted_data[n // 2]
else:
    median = (sorted_data[n // 2 - 1] + sorted_data[n // 2]) / 2
print("Median:", median)

freq = {}
for i in data:
    freq[i] = freq.get(i, 0) + 1
minn=0
mode=0
for a,b in freq.items():
    if b>minn:
        minn=b
        mode=a
print("Mode:",mode)

mean_diff_squared = [(x - mean) ** 2 for x in data]
sd = (sum(mean_diff_squared) / (n-1)) ** 0.5
print("Standard Deviation:",round(sd,2))

plt.boxplot(data)
plt.title("Box Plot of the Data")
plt.ylabel("Values")
plt.show()

Median: 40.5

#22) Normalization
import matplotlib.pyplot as plt

l=eval(input("List of numbers:"))
minn=min(l)
maxx=max(l)
data=[]
for i in l:
    normalized=((i - minn) / (maxx- minn))
    data.append(round(normalized,2))
print("Original numbers:",l)
print("Normalized numbers:", data)

plt.plot(l, label='Original Data', marker='o')
plt.plot(data, label='Normalized Data', marker='x', color='red')
plt.xlabel('Index')
plt.ylabel('Value')
plt.title('Normalization of Data (Original vs Normalized)')
plt.legend()
plt.grid()
plt.tight_layout()
plt.show()
